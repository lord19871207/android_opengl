#version 300 es

uniform sampler2D sTexture;

//uniform float uApertureDiameter;
//uniform float uFocalLength;
//uniform float uPlaneInFocus;
//uniform float uSensorHeight;

uniform float uCamera1;
uniform float uCamera2;

in vec2 vTexPosition;

layout (location = 0) out vec3 outFragColor;
layout (location = 1) out vec2 outFragCoC;

void main() {
    vec4 sample = texture(sTexture, vTexPosition);
    float coc = (uCamera1 / sample.a) - uCamera2;
    outFragCoC = sample.a <= 0.0 ? vec2(1.0, 0.0) : clamp(vec2(-coc, coc), 0.0, 1.0);
    outFragColor = sample.rgb;
}
